## Usage

```tsx
import {
  TagSelect,
  TagSelectActions,
  TagSelectContent,
  TagSelectTags,
} from "@/components/ui/tag-select";
```

```tsx
<TagSelect>
  <TagSelectContent>
    <TagSelectTags />
    <TagSelectActions />
  </TagSelectContent>
</TagSelect>
```

## OpenUI Integration

Register `TagSelect` with `defineComponent`, describe its model-generated props with Zod, and include it in a library.

```tsx showLineNumbers
import {
  TagSelect,
  TagSelectActions,
  TagSelectContent,
  TagSelectTags,
} from "@/components/ui/tag-select";

const OpenUITagSelect = defineComponent({
  name: "TagSelect",
  description:
    "Colored tag selector with single or multiple selection and color variants.",
  props: z.object({}),
  component: () => (
    <TagSelect>
      <TagSelectContent>
        <TagSelectTags />
        <TagSelectActions />
      </TagSelectContent>
    </TagSelect>
  ),
});

export const tag_selectLibrary = createLibrary({
  root: "TagSelect",
  components: [OpenUITagSelect],
});
```

For richer generated compositions, define the child primitives separately and connect them with `.ref` fields. See [Defining Components](https://www.openui.com/docs/openui-lang/defining-components) in the OpenUI documentation.
