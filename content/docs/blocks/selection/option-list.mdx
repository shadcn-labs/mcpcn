## Usage

```tsx
import {
  OptionList,
  OptionListActions,
  OptionListContent,
  OptionListOptions,
} from "@/components/ui/option-list";
```

```tsx
<OptionList>
  <OptionListContent>
    <OptionListOptions />
    <OptionListActions />
  </OptionListContent>
</OptionList>
```

## OpenUI Integration

Install the OpenUI React packages.

```bash
pnpm add @openuidev/react-lang zod
```

Add `OptionList` to a component library and map it to the installed component.

```tsx showLineNumbers
import { createLibrary, defineComponent } from "@openuidev/react-lang";
import { z } from "zod/v4";

import {
  OptionList,
  OptionListActions,
  OptionListContent,
  OptionListOptions,
} from "@/components/ui/option-list";

const OpenUIOptionList = defineComponent({
  name: "OptionList",
  description:
    "Tag-style option selector with single or multiple selection modes.",
  props: z.object({}),
  component: () => (
    <OptionList>
      <OptionListContent>
        <OptionListOptions />
        <OptionListActions />
      </OptionListContent>
    </OptionList>
  ),
});

export const option_listLibrary = createLibrary({
  root: "OptionList",
  components: [OpenUIOptionList],
});
```

For richer generated compositions, define the child primitives separately and connect them with `.ref` fields. See [Defining Components](https://www.openui.com/docs/openui-lang/defining-components) in the OpenUI documentation.

## json-render Integration

Install the json-render React packages.

```bash
pnpm add @json-render/core @json-render/react zod
```

Add `OptionList` to a component catalog and map it to the installed component.

```tsx
import { defineCatalog } from "@json-render/core";
import { defineRegistry } from "@json-render/react";
import { schema } from "@json-render/react/schema";
import { z } from "zod";

import {
  OptionList,
  OptionListActions,
  OptionListContent,
  OptionListOptions,
} from "@/components/ui/option-list";

export const catalog = defineCatalog(schema, {
  components: {
    OptionList: {
      props: z.object({}),
      description:
        "Tag-style option selector with single or multiple selection modes.",
    },
  },
});

export const { registry } = defineRegistry(catalog, {
  components: {
    OptionList: () => (
      <OptionList>
        <OptionListContent>
          <OptionListOptions />
          <OptionListActions />
        </OptionListContent>
      </OptionList>
    ),
  },
});
```

Use `catalog.prompt()` to constrain model output, then render the generated spec with the json-render React `Renderer`. See the [json-render Quick Start](https://json-render.dev/docs/quick-start) for the provider and streaming setup.
